## 🔧 Implementation Strategy

### **Pattern to Follow**

For every action (view, create, edit, delete):

```typescript
// 1. Get current role from store
const { currentRole } = useStore();

// 2. Check permission
const permCheck = permissionChecker.hasPermission(currentRole, {
  permission: 'edit',
  resource: 'connection',
  atScope: connection.scopePath  // Optional: check at specific scope
});

// 3. Handle result
if (!permCheck.allowed) {
  // Option A: Hide the action
  return null;

  // Option B: Disable the action
  <Button disabled title={permCheck.reason}>Edit</Button>

  // Option C: Show request access
  <PermissionRequestModal requirement={permCheck.requirement} />
}

// 4. Show the action
<Button onClick={handleEdit}>Edit</Button>
```

## 🎨 UI Patterns

### **Pattern 1: Action Buttons**
```typescript
// Before RBAC
<Button onClick={handleDelete}>Delete</Button>

// After RBAC
{canDelete ? (
  <Button onClick={handleDelete}>Delete</Button>
) : (
  <Tooltip content="Requires admin role">
    <Button disabled>Delete</Button>
  </Tooltip>
)}
```

### **Pattern 2: Menu Filtering**
```typescript
// Before RBAC
const actions = ['edit', 'delete', 'duplicate', 'export'];

// After RBAC
const actions = [
  { id: 'edit', requires: 'edit' },
  { id: 'delete', requires: 'delete' },
  { id: 'duplicate', requires: 'create' },
  { id: 'export', requires: 'view' }
].filter(action =>
  permissionChecker.hasPermission(currentRole, {
    permission: action.requires
  }).allowed
);
```

### **Pattern 3: Data Filtering by Scope**
```typescript
// Before RBAC
const visibleConnections = connections;

// After RBAC
const visibleConnections = connections.filter(conn => {
  // Check if connection is within user's scope
  return isWithinScope(userScope, conn.scopePath);
}).filter(conn => {
  // Apply resource filter
  return matchesFilter(resourceFilter, conn, currentUser);
});
```

### **Pattern 4: Permission Banner**
```typescript
// Show banner when user lacks permission
{!canEdit && (
  <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-3">
    <div className="flex items-center gap-2">
      <Lock className="h-4 w-4 text-yellow-600" />
      <p className="text-sm text-yellow-800">
        You need <strong>edit</strong> permission to modify this resource.
      </p>
      <Button size="sm" variant="outline">Request Access</Button>
    </div>
  </div>
)}
```
